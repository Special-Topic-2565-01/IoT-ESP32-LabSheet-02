## 63030091 สิริภักศร รัตนคช

*ข้อที่ 1 วิ่งไปสุดปลายทางแล้ววิ่งกลับ (ping-pong)*

https://youtu.be/upfJlS3SApo
```c
    #include <stdio.h>
    #include <stdbool.h>
    #include <unistd.h>
    #include "driver/gpio.h"
    uint32_t ON_time = 500000;
    uint32_t OFF_time = 0;
    uint8_t LEDIO[16] = { 23, 22, 1, 3, 21, 19, 18, 5, 18, 19, 21, 3, 1, 22, 23 };
    void app_main(void)
    {
        for (uint8_t i = 0; i < 8; i++)
        {
            gpio_reset_pin(LEDIO[i]);
        }
        for (uint8_t i = 0; i < 8; i++)
        {
            gpio_set_direction(LEDIO[i], GPIO_MODE_OUTPUT);
        }
        uint8_t ledNum = 0;
        while (true)
        {
            gpio_set_level(LEDIO[ledNum], 1);
            usleep(ON_time);
            gpio_set_level(LEDIO[ledNum], 0);
            usleep(OFF_time);
            ledNum++;
            if (ledNum == 16)
                ledNum = 0;
        }
    }
```

        for (uint8_t i = 0; i < 8; i++)
        {
            gpio_reset_pin(LEDIO[i]);
        }
        for (uint8_t i = 0; i < 8; i++)
        {
            gpio_set_direction(LEDIO[i], GPIO_MODE_OUTPUT);
        }
        uint8_t ledNum = 0;
        while (true)
        {
            gpio_set_level(LEDIO[ledNum], 1);
            usleep(ON_time);
            ledNum++;
            if (ledNum == 9)
            {
            	for (int i = 0; i < 9; i++)
            	{
            		gpio_set_level(LEDIO[ledNum], 0);
            		usleep(ON_time);
            		ledNum--;
            	}
            }
        }
    }
